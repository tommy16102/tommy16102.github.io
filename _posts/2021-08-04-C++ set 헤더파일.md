# Set
> `#include <set>`
> 중복이 안되며(multiset), 자동으로 정렬이 된다.


```
#include <iostream>
#include <set>
using namespace std;

int main() {
	set<int> s;
	s.insert(1);
	s.insert(1);
	s.insert(3);
	s.insert(5);
	s.insert(4);
	s.insert(2);
	s.insert(2);

	auto it = s.find(4);
	if (it != s.end()) cout << *it << endl;
	//4 출력

	for (auto it = s.begin();it != s.end();it++) {
		cout << *it;
	}
	//12345 출력(중복 x, 오름차순 정렬 o)

	s.erase(s.begin());
	//2345

	s.size();
	//4
}

```
